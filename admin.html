import { auth } from "./firebase.js"; import { onAuthStateChanged, signOut } from "https://www.gstatic.com/firebasejs/12.0.0/firebase-auth.js"; onAuthStateChanged(auth, (user) => { if (!user) { window.location.href = "login.html"; } }); window.logout = async function(){ try { await signOut(auth); window.location.href = "login.html"; } catch(error){ alert(error.message); } };
